VOTE CALCULATORresult="G"numGvotes = 0numTvotes = 0while (result.upper() == "G" or result.upper() == "T"):   result = input("Pls enter your vote (G/T/q to quit)")  if (result.upper() == "G"):    numGvotes += 1  elif (result.upper() == "T"):    numTvotes += 1print("Total number of votes for G is:" +str(numGvotes))print("Total number of votes for T is:" +str(numTvotes))if (numGvotes>numTvotes):  print("G WON!")elif (numTvotes>numGvotes):  print("T WON!")else:  print("They have equal votes")
